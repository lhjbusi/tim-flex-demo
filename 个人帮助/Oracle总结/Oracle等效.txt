DECLARE
   EMP_ROWS EMP%Rowtype;
       CURSOR EMP_CUR IS SELECT * FROM EMP WHERE EMPNO >= 7369;
BEGIN
     OPEN EMP_CUR;
     LOOP
         FETCH EMP_CUR INTO EMP_ROWS;
         EXIT WHEN EMP_CUR%NOTFOUND;
         DBMS_OUTPUT.put_line('EMPNO:  ' || EMP_ROWS.EMPNO || '    EMP.JOB:   ' || EMP_ROWS.JOB);
     END LOOP;
     CLOSE EMP_CUR;
END;


DECLARE
       CURSOR TEST IS SELECT * FROM EMP WHERE EMPNO>=7369;
BEGIN
     FOR EMP_TEST IN TEST LOOP
         DBMS_OUTPUT.put_line('EMP.EMPNO:   ' || EMP_TEST.EMPNO);
     END LOOP;     
END;
